##create input directory and put the input file
hdfs dfs -mkdir /input 
hdfs dfs -put <input file> /input

##cross check if the input file is copied over to hdfs
hdfs dfs -ls /input
hdfs dfs -cat /input/<input file>

hadoop jar PokerMissingCards.jar PokerMissingCards /input/<input file> /output

##To check the output folder:

hdfs dfs -ls /output

hdfs dfs -cat /output/part-r-00000
